create database livraria;
use livraria
create table categoria(
    id integer not null primary key,
    descricao varchar(300),
    livros varchar(13)
    );
create table livro(
    isbn varchar(13) not null primary key,
    titulo varchar(100),
    ano year,
    editora varchar(50),
    autor integer,
    categoria integer
    );
create table autor(
    id integer not null primary key,
    nacionalidade varchar(30),
    nome varchar(100),
    livros varchar(13)
    );
alter table livro
    add foreign key (autor) references autor(id),
    add foreign key (categoria) references categoria(id);
alter table categoria
    add foreign key (livros) references livro(isbn);
alter table autor
    add foreign key (livros) references livro(isbn);
:)\g
